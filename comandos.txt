git init                            // iniciar pasta git
git status                          // ve o status
git add                             // adiciona na stash
ls -a                               // mostra pastas e arquivos escondidos
git commit -m "adicionei comandos"  // coloca nomes do commit
git log                             // mostra os commits feitos
git checkout 'onde tu quer ir'      // muda de branch
git checkout -b 'onde tu quer ir'      // muda de branch criando uma nova
git push origin master              // coloca no github
